\section{ILP Formulation}

Let $T$ be the set of tasks, $J$ the set of jobs, $C$ the set of cores, and $K$ the set of clusters.
Each job $j \in J$ belongs to a parent task $\tau(j) \in T$.
Let $D_{\mathrm{job}} \subseteq J \times J$ denote job-level precedence dependencies and
$D_{\mathrm{task}} \subseteq T \times T$ denote task-level communication dependencies.
Each task $t \in T$ may only be assigned to an eligible core in $E_t \subseteq C$.
For cluster $k \in K$, let $C_k \subseteq C$ denote its cores.

\paragraph{Model}
\begin{align}
\min \quad
&
\lambda_{\mathrm{core}} \sum_{c \in C} o_c^{\mathrm{core}}
+
\lambda_{\mathrm{cluster}} \sum_{k \in K} o_k^{\mathrm{cluster}}
+
\sum_{(t_1,t_2)\in D_{\mathrm{task}}}
\sum_{c \in E_{t_1}}
\sum_{d \in E_{t_2}}
q_{cd}\, z_{t_1t_2cd}
\end{align}

\paragraph{Subject to:\\}

\textit{Task assignment}
\begin{align}
\sum_{c \in E_t} y_{tc} &= 1
&& \forall t \in T \\
y_{tc} &= 0
&& \forall t \in T,\; c \in C \setminus E_t
\end{align}

\textit{Partitioned job assignment}
\begin{align}
x_{jc} &= y_{\tau(j)c}
&& \forall j \in J,\; c \in C
\end{align}

\textit{Timing}
\begin{align}
s_j &\ge r_j
&& \forall j \in J \\
s_j &\le r_j + \mathrm{jitter}
&& \text{for applicable root jobs} \\
f_j &= s_j + \sum_{c \in C} p_j \alpha_c x_{jc}
&& \forall j \in J \\
f_j &\le d_j
&& \forall j \in J \text{ with } d_j \neq \emptyset
\end{align}

\textit{Precedence}
\begin{align}
s_{j'} &\ge f_j
&& \forall (j,j') \in D_{\mathrm{job}}
\end{align}

\textit{Core memory}
\begin{align}
\sum_{t \in T} m_t y_{tc}
&\le B_c^{\mathrm{core}} + o_c^{\mathrm{core}}
&& \forall c \in C
\end{align}

\textit{Cluster memory}
\begin{align}
\sum_{t \in T} \sum_{c \in C_k} m_t y_{tc}
&\le B_k^{\mathrm{cluster}} + o_k^{\mathrm{cluster}}
&& \forall k \in K
\end{align}

\textit{Non-overlap on cores}
\begin{align}
s_{j'}
&\ge
f_j - M\left(3 - u_{jj'} - x_{jc} - x_{j'c}\right)
&& \forall j,j' \in J,\; j<j',\; c \in C \\
s_j
&\ge
f_{j'} - M\left(2 + u_{jj'} - x_{jc} - x_{j'c}\right)
&& \forall j,j' \in J,\; j<j',\; c \in C
\end{align}

\textit{Communication linearization}
\begin{align}
z_{t_1t_2cd}
&\ge
y_{t_1c} + y_{t_2d} - 1
&& \forall (t_1,t_2)\in D_{\mathrm{task}},\;
c\in E_{t_1},\;
d\in E_{t_2}
\end{align}

\paragraph{Domains}
\begin{align}
y_{tc} &\in \{0,1\}
&& \forall t \in T,\; c \in C \\
x_{jc} &\in \{0,1\}
&& \forall j \in J,\; c \in C \\
u_{jj'} &\in \{0,1\}
&& \forall j,j' \in J,\; j<j' \\
z_{t_1t_2cd} &\in \{0,1\}
&& \forall (t_1,t_2)\in D_{\mathrm{task}},\;
c\in E_{t_1},\;
d\in E_{t_2} \\
s_j,\; f_j &\in [0,M]
&& \forall j \in J \\
o_c^{\mathrm{core}} &\ge 0
&& \forall c \in C \\
o_k^{\mathrm{cluster}} &\ge 0
&& \forall k \in K
\end{align}
